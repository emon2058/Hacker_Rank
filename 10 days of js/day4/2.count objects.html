    function getCount(objects) {
    let sameValue=0;
    for(let i=0;i<objects.length;i++){ //here length is 5
       if(objects[i].x==objects[i].y){ //objects[0].x will be 1 and objects[0].y will be 1
        sameValue +=1;
        }
    }
  return sameValue;
}
