        class Rectangle {
        constructor(w, h) {
            this.w = w;
            this.h = h;
        }
      }
        Rectangle.prototype.area=function(){
          return this.w*this.h;
        }

      class Square extends Rectangle{
        constructor(length){
          super(length);
          this.h=length;
          this.w=length;
        }
      }
      if (JSON.stringify(Object.getOwnPropertyNames(Square.prototype)) === JSON.stringify([ 'constructor' ])) {
          const rec = new Rectangle(3, 4);

          console.log(-1);
